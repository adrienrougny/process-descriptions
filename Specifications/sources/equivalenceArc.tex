\subsection{Glyph: \glyph{Equivalence arc} }\label{sec:equivalenceArc}

\corr{\glyph{Equivalence arc} is the arc used to link a tag of a submap to an EPN of the main map or of the submap?}{An \glyph{equivalence arc} is used to link a \glyph{tag} (\sect{tag}) or a \glyph{submap terminal} (\sect{submapTerminal}) to the \glyph{EPN} (\sect{EPNs}) or \glyph{compartment} (\sect{compartment}) it refers to.}

% \glyph{Equivalence arc} is the arc used to represent the fact that all entities
% marked by a \glyph{tag} are equivalent.

\begin{glyphDescription}

\glyphSboTerm
Not applicable.

\glyphOrigin
\corr{Any}{One} \glyph{EPN} (\sect{EPNs}) \add{or \glyph{compartment} (\sect{compartment})}.

\glyphTarget
\add{One} \glyph{tag} (\sect{tag}) \add{or \glyph{submap terminal} (\sect{submapTerminal})}.

\corr{\glyphEndPoint}{\glyphSymbol} No particular symbol is used to represent an \glyph{equivalence arc}\add{, as shown in \fig{equivalenceArc}}.
 \end{glyphDescription}
